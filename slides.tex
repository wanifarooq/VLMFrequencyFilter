\section{Theoretical Framework}

\begin{frame}{The Task-Specific Filter $\Wt$}
\begin{columns}[T,onlytextwidth]
\column{0.52\linewidth}
\small
\begin{enumerate}
  \item Vision tokens $\{v_{ij}\}$ on an $H\times W$ grid.
  \item Cross-modal attention at layer group $\ell$ gives spatial map $a_t^{(\ell)}$.
  \item 2D FFT, square, radially bin into $B$ linear bands.
  \item Average over images; L1-normalise:
\end{enumerate}
\[
\Wt^{(\ell)} = \frac{\mathbb{E}_I[\,|\mathcal{F}_2\, a_t^{(\ell)}(I)|^2(\omega)\,]}{\sum_{\omega'}\mathbb{E}_I[\ldots]}
\]

\vspace{0.3em}
\textbf{Effective bandwidth via IPR} (L1-normalised probability):
\[
\Gt = \left(\sum_\omega \Wt^2\right)^{-1},\quad 1\le\Gt\le B
\]

\begin{frame}{The Overlap Law}
\vspace{0.3em}
\begin{columns}[T,onlytextwidth]
\column{0.50\linewidth}
\textbf{Perturbation spectral signature:}
\[
\DF_p(\omega) = \mathbb{E}_I\!\left[\,\bigl|\,|\mathcal{F}_2 I_p|^2 - |\mathcal{F}_2 I|^2\,\bigr|\,\right]_\omega

Model-independent; purely a property of $p$.

\textbf{Central claim:}
\[
\mathbb{E}[\,\Delta\mathrm{acc}(t,p)\,] \;\propto\; \Spred(t,p)

\begin{frame}{Regression Specification}
\textbf{Residualise, then horse-race (primary target: loglik volatility)}
\begin{align*}
\Cres &= \Csem - \hat\Csem(\Cprompt) \\[0.2em]
y_{i,t} &= \alpha_i + \beta_{\Csem}\Csem(t) + \beta_{\Cprompt}\Cprompt(t) + \beta_H H_{\mathrm{opt}}(t) + \varepsilon_{i,t}
\end{align*}

\vspace{0.5em}
\begin{columns}[T,onlytextwidth]
\column{0.48\linewidth}
\begin{block}{\textcolor{pnavy}{Primary}}
\small
Log-likelihood volatility\\
$\mathbb{V}_p[\log p_\theta(a^\star\mid I_p,t)]$\\
continuous, model-native
\end{block}
\column{0.48\linewidth}
\begin{block}{\textcolor{pnavy}{Secondary}}
\small
Gated accuracy drop\\
clean-correct $\to$ perturbed-wrong\\
thresholded, binary, confirmatory
\end{block}
\end{columns}

\vspace{0.4em}
\textbf{Paired mirror t-tests} (L1$\leftrightarrow$L5, L2$\leftrightarrow$L6, L3$\leftrightarrow$L7, L4$\leftrightarrow$L8) identify the $\Cprompt$ effect without regression assumptions.
\end{frame}

\begin{frame}{Limitations}
\begin{columns}[T,onlytextwidth]
\column{0.48\linewidth}
\begin{block}{\textcolor{pnavy}{What's not yet validated}}
\small
\begin{itemize}
  \item Single model (Qwen3-VL-2B); LLaVA-NeXT / InternVL replication in progress.
  \item Severity-1 only in the main run.
  \item Task 4 (cutoff) reported qualitatively.
\end{itemize}
\end{block}

\column{0.48\linewidth}
\begin{block}{\textcolor{pnavy}{What's inherent}}
\small
\begin{itemize}
  \item $\Wt$ from attention $\Rightarrow$ correlational lower bound.
  \item $\Spred$ is population-level ($r \approx 0.55$ grouped), not per-sample ($r \approx 0.02$).
  \item $\Csem$ is equal-weighted by design; sign invariant under any positive reweighting, magnitudes are not.
  \item Mirror controls are synthetic; human-judged paraphrases are a natural extension.
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Conclusion}
\keybox{Can we predict, \emph{before} deploying a VLM, how badly it will break on a given task under a given corruption profile?}
\vspace{0.3em}

\textbf{We give one answer.}
\begin{itemize}\setlength\itemsep{0.25em}
  \item The text is a \textcolor{pnavy}{\textbf{physical spectral operator}} $\Wt$.
  \item Reasoning is an \textcolor{paccent}{\textbf{upward migration}} of attention mass in the Fourier basis.
  \item $\Spred = \sum_\omega |\Wt|^2|\DF|^2$ predicts damage at $r \approx 0.55$ with \emph{no} target-side fitting.
  \item A \textbf{tug-of-war} on loglik volatility: $\tilde\beta(\Cres) = +0.29$, $\tilde\beta(\Cprompt) = -0.32$, $R^2 = 0.22$.
  \item \textbf{Mechanism:} early-layer lexical anchoring $\to$ late-layer upward migration.
\end{itemize}

\vspace{0.4em}
\textcolor{pnavy}{\textbf{Immediate payoff:}} a deployment audit tool that turns silent failure into an engineerable quantity --- plus training-free interventions.

\vspace{0.2em}
\textcolor{pnavy}{\textbf{Broader payoff:}} Fourier, compositional, and mechanistic accounts become three facets of a single falsifiable law.
\end{frame}

\begin{frame}{Backup: $\Csem$ Axes}
\[
\Csem = E + A + R + O + D + \ln(1 + G_{\mathrm{amb}})
\]
\begin{itemize}
  \item $E$ --- entity refs (GQA \texttt{select}/\texttt{query})
  \item $A$ --- attribute refs (GQA \texttt{query\_attr}, CLEVR)
  \item $R$ --- relation refs (GQA / CLEVR \texttt{relate})
  \item $O$ --- reasoning ops (NMN primitives)
  \item $D$ --- program depth (CLEVR's key difficulty axis)
  \item $G_{\mathrm{amb}}$ --- grounding ambiguity (log form, ReferIt)
\end{itemize}
\end{frame}

\begin{frame}{Backup: $\DZ$ --- Mean L2 Drift}
\[
\DZ(t,p)^{(\ell)} = \frac{1}{N^{(\ell)}} \sum_{n=1}^{N^{(\ell)}} \bigl\| z_n^{\mathrm{pert},(\ell)} - z_n^{\mathrm{clean},(\ell)} \bigr\|_2 .
\]
\begin{itemize}
  \item Mean $\ell_2$ over full multimodal sequence.
  \item Language tokens aligned positionally.
  \item Vision-token spans aligned with patch-grid-aware resampling.
\end{itemize}
\end{frame}
